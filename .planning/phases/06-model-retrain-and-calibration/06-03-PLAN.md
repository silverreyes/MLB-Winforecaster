```python
import pandas as pd
import pytest
import numpy as np
from sklearn.metrics import brier_score_loss

def test_brier_comparison_has_four_groups():
    """MDL-03: Comparison table has all four groups."""
    comparison = pd.read_csv("data/results/brier_comparison.csv")
    groups = set(comparison['group'].values)
    assert groups == {'v2_sp_enhanced', 'v2_team_only', 'v1', 'kalshi'}, f"Groups: {groups}"

def test_brier_comparison_same_n_games():
    """MDL-03: All groups compared on identical game set (same n_games)."""
    comparison = pd.read_csv("data/results/brier_comparison.csv")
    n_games_values = comparison['n_games'].unique()
    assert len(n_games_values) == 1, f"Different n_games across groups: {comparison[['group','n_games']].to_dict()}"

def test_brier_comparison_valid_scores():
    """MDL-03: Brier scores are valid floats in reasonable MLB range."""
    comparison = pd.read_csv("data/results/brier_comparison.csv")
    for _, row in comparison.iterrows():
        assert 0 <= row['brier_score'] <= 0.5, f"{row['group']} Brier {row['brier_score']} out of range"

def test_brier_comparison_columns():
    """MDL-03: Comparison CSV has required columns."""
    comparison = pd.read_csv("data/results/brier_comparison.csv")
    required = {'group', 'model_name', 'brier_score', 'n_games'}
    assert required.issubset(set(comparison.columns)), f"Missing columns: {required - set(comparison.columns)}"
```

```python
print(f"V1 predictions: {preds_v1.shape[0]} rows")

game_keys = preds_v2[['game_date', 'home_team', 'away_team']].drop_duplicates()
```

**Cell 2 - Find Common Game Set (Intersection):**
```python
# Create game_key for joining
def make_key(df):
    return df.apply(lambda r: (str(r['game_date'])[:10], r['home_team'], r['away_team']), axis=1)

# Get unique games per source
v2_games = set(make_key(preds_v2.drop_duplicates(['game_date','home_team','away_team'])))
v1_games = set(make_key(preds_v1.drop_duplicates(['game_date','home_team','away_team'])))
kalshi_games = set(make_key(kalshi_prices))

# Intersection = games present in ALL sources
common_games = v2_games & v1_games & kalshi_games
print(f"Common games: {len(common_games)} (v2: {len(v2_games)}, v1: {len(v1_games)}, kalshi: {len(kalshi_games)})")
```

**Cell 3 - Compute Brier Scores for Each Group:**
Use the BEST model from each group (lowest Brier) for the comparison row, but also show all models:

```python
# Filter to common games
def filter_common(df):
    keys = make_key(df)
    return df[keys.isin(common_games)]

# V2 SP_ENHANCED: best model (likely LR based on v1 patterns)
v2_sp = filter_common(preds_v2[preds_v2['feature_set'] == 'sp_enhanced'])
v2_team = filter_common(preds_v2[preds_v2['feature_set'] == 'team_only'])
v1_core = filter_common(preds_v1)
kalshi_common = filter_common(kalshi_prices)

comparison_rows = []

# V2 SP_ENHANCED - per model
for model in ['lr', 'rf', 'xgb']:
    subset = v2_sp[v2_sp['model_name'] == model]
    brier = brier_score_loss(subset['home_win'], subset['prob_calibrated'])
    comparison_rows.append({
        'group': 'v2_sp_enhanced', 'model_name': model,
        'brier_score': round(brier, 4), 'n_games': len(subset)
    })

# V2 TEAM_ONLY - per model
for model in ['lr', 'rf', 'xgb']:
    subset = v2_team[v2_team['model_name'] == model]
    brier = brier_score_loss(subset['home_win'], subset['prob_calibrated'])
    comparison_rows.append({
        'group': 'v2_team_only', 'model_name': model,
        'brier_score': round(brier, 4), 'n_games': len(subset)
    })

# V1 - per model
for model in ['lr', 'rf', 'xgb']:
    subset = v1_core[v1_core['model_name'] == model]
    brier = brier_score_loss(subset['home_win'], subset['prob_calibrated'])
    comparison_rows.append({
        'group': 'v1', 'model_name': model,
        'brier_score': round(brier, 4), 'n_games': len(subset)
    })

# Kalshi market
# Kalshi "probability" is kalshi_open_price (already in [0,1])
# Need to join with outcomes
kalshi_with_outcomes = kalshi_common.merge(
    v2_sp[['game_date','home_team','away_team','home_win']].drop_duplicates(),
    on=['game_date','home_team','away_team']
)
kalshi_brier = brier_score_loss(kalshi_with_outcomes['home_win'], kalshi_with_outcomes['kalshi_open_price'])
comparison_rows.append({
    'group': 'kalshi', 'model_name': 'market',
    'brier_score': round(kalshi_brier, 4), 'n_games': len(kalshi_with_outcomes)
})

comparison_df = pd.DataFrame(comparison_rows)
comparison_df.to_csv("data/results/brier_comparison.csv", index=False)
print("\nBrier Score Comparison (2025, identical game set):")
print(comparison_df.to_string(index=False))
```

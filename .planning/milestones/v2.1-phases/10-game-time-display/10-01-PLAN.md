      ```typescript
      function formatGameTime(isoString: string | null): string {
        if (!isoString) return 'Time TBD';
        try {
          const date = new Date(isoString);
          if (isNaN(date.getTime())) return 'Time TBD';
          return date.toLocaleTimeString('en-US', {
            hour: 'numeric',
            minute: '2-digit',
            timeZone: 'America/New_York',
          }) + ' ET';
        } catch {
          return 'Time TBD';
        }
      }
      ```
      This uses `Intl.DateTimeFormat` (via `toLocaleTimeString`) with `timeZone: "America/New_York"` per GMTIME-02. Output format: "7:05 PM ET". The `hour: 'numeric'` option avoids leading zero (7 not 07). The `minute: '2-digit'` ensures ":05" not ":5".
